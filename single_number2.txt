/**
 * @input A : Read only ( DON'T MODIFY ) Integer array
 * @input n1 : Integer array's ( A ) length
 * 
 * @Output Integer
 */
int singleNumber(const int* A, int n1) 
{
    int j=0,i,r=0,count=0;
    for(i=0;i<31;i++)
    {
        count=0;
        for(j=0;j<n1;j++)
        {
            if(A[j]&(1<<i))
            count++;
            count=count%3;
        }
        if(count)
        r=r|(1<<i);
    }
    return r;
}
